version: '3.8'

services:
  # --- SEU BACKEND/FRONTEND NODE.JS ---
  kns-noc-app:
    container_name: kns-noc-app
    build: .
    ports:
      - "5000:5000"
    environment:
      - NODE_ENV=production
    volumes:
      - .:/app
      - /app/node_modules
    restart: always

  # --- O MONSTRO: GRAFANA CONFIGURADO ---
  kns-grafana:
    container_name: kns-grafana
    image: grafana/grafana:latest
    ports:
      - "3000:3000"
    environment:
      # 1. Instala o Plugin do Zabbix automaticamente ao iniciar
      - GF_INSTALL_PLUGINS=alexanderzobnin-zabbix-app
      
      # 2. Permite que o Grafana seja colocado num Iframe (dentro do seu site)
      - GF_SECURITY_ALLOW_EMBEDDING=true
      
      # 3. Permite acesso Anônimo (visitante vê sem logar)
      - GF_AUTH_ANONYMOUS_ENABLED=true
      - GF_AUTH_ANONYMOUS_ORG_ROLE=Viewer
      - GF_AUTH_DISABLE_LOGIN_FORM=false
      
      # 4. Libera acesso de qualquer origem (CORS - Evita erro de navegador)
      - GF_SECURITY_COOKIE_SAMESITE=none
    restart: always
