# Usa uma imagem leve do Node.js (Alpine Linux)
FROM node:18-alpine

# Cria a pasta de trabalho dentro do container
WORKDIR /app

# Copia os arquivos de dependência primeiro (para cache eficiente)
COPY package*.json ./

# Instala as dependências do projeto
RUN npm install --production

# Copia o restante do código (server.js, index.html, etc)
COPY . .

# Expõe a porta que usamos no server.js
EXPOSE 5000

# Comando para iniciar a aplicação
CMD ["node", "server.js"]
